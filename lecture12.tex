\begin{definition}
    A point $x$ is a limit point (or cluster point, or accumulation point) of a set $A$ if every $\epsilon$-neighborhood $V_\epsilon(x)$ of $x$ intersects the set $A$ in some point other than $x$.  
\end{definition}

\begin{theorem}
    A point $x$ is a limit point of a set $A$ if and only if $x = \lim a_n$ for some sequence $\{a_n\}\subset A$ such that $a_n \neq x$ for all $n \in \mathbb{N}$.
\end{theorem}
