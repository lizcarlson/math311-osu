\begin{exercise}
    Let $A\subseteq \mathbb{R}$ be a bounded set.  Prove that $\inf{(A)} = - \sup{(-A)}$.
\end{exercise}
\newpage
\begin{exercise}
    Provide an example of nested intervals of the form $[a_n,b_n)$ (i.e., $a_n \leq x < b_n$ for all $x \in [a_n,b_n)$) such that $[a_{n+1},b_{n+1}) \subseteq [a_n,b_n)$ and $\bigcap\limits_{n=1}^\infty [a_n,b_n) = \emptyset$.  No need to prove your answer, simply give a brief explanation of the intuition.
\end{exercise}
