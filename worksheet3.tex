\begin{exercise}
	Find examples of sequences that verify the following statements, or prove why the following statements are false:
\begin{enumerate}
	\item sequences $\{a_n\}$ and $\{b_n\}$ that diverge, but $\{a_n+b_n\}$ converges.
	\item a sequence $\{a_n\}$ that converges and a sequence $\{b_n\}$ that diverges, but $\{a_n+b_n\}$ converges.
	\item an unbounded sequence $\{a_n\}$ and a convergence sequence $\{b_n\}$ such that $\{a_n-b_n\}$ is bounded.
\end{enumerate}
\end{exercise}

\begin{exercise}
	Consider the sequence $\{a_n\} = \left\{1+\frac{1}{n^2}\right\}$, and let \[b_n = \{0,1,0,1,0,1,0,1,1, 2, \ldots , n-8, \ldots\}.\] Can the Algebraic Limit Theorem be used to calculate the limit of $\left\{\frac{a_n}{b_n}\right\}$?  Why or why not?
\end{exercise}
