    \begin{exercise}{(Exercise 1.4.2)}
        Let $\mathbb{I}$ stand for the set of irrational numbers.  Show that if $a \in \mathbb{Q}$ and $t \in I$, then $a+t \in I$ and $at \in I$ as long as $a \neq 0$.
     \end{exercise}
