    \begin{exercise}
        Let $S$, $T$ be nonempty bounded subsets of $\mathbb{R}$.
        \begin{enumerate}
            \item What can you say about $S$ if $\inf(S) = \sup(S)$? (Provide a rigorous answer using a proof.)
            \item Prove that if $S \subseteq T$, then $\inf(T) \leq \inf(S) \leq \sup(S) \leq \sup(T)$.
        \end{enumerate}
    \end{exercise}
    \begin{exercise}
        Let $a,b,c, \in \mathbb{R}$.  Prove that the following are equivalent for the real numbers
        
        \phantom{blahblahblah} \begin{enumerate*}
            \item $|a-b| < c$ \phantom{blah}
            \item $b-c < a < b+c$ \phantom{blah}
            \item $a \in (b-c, b+c)$.
        \end{enumerate*}
    \end{exercise} 
    \begin{exercise}
        Prove that the sequence $\left\{\frac{n+1}{n}\right\}$ converges to $1$.
    \end{exercise}
    \begin{exercise}
        Show that if $\lim b_n = b$, then $\lim |b_n| = |b|$.
    \end{exercise}
    \begin{exercise}
        Prove whether or not the following sets are nonempty:
        
        \phantom{blahblahblahblah}
        \begin{enumerate*}
            \item $\bigcap\limits_{n=1}^\infty \left[-\frac1n, \frac1n\right]$ \phantom{blah}
            \item $\bigcap\limits_{n=1}^\infty \left(-\frac1n, \frac1n\right)$ \phantom{blah}
            \item $\bigcap\limits_{n=1}^\infty \left( 0, \frac1n\right)$
        \end{enumerate*}
    \end{exercise}

   \textit{\textbf{Choose one of the following problems to submit for grading:}}
    \begin{exercise}
        Prove that if a sequence $\{a_n\}$ converges to $a$, its limit is unique.
    \end{exercise}

    \begin{exercise}
        Consider 3 sequences $\{a_n\}, \{b_n\}, \{s_n\}$ such that $a_n\leq s_n\leq b_n$ for all $n \in \mathbb{N}$.  Suppose $\lim a_n = \lim b_n =s$.  Show that $\lim s_n = s$.  
        
        (Note: there are at least three different ways to prove this based on what we have learned in class. \underline{You will get extra credit if you provide two different proofs.})
    \end{exercise}
