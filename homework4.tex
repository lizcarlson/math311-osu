\begin{exercise}
Let $\{s_n\}$ be a sequence.
    \begin{enumerate}
        \item Suppose 
        \begin{align*}
            |s_{n+1} - s_n| < 2^{-n} \quad \text{ for all } \;  n \in \mathbb{N}.
        \end{align*}
        Prove that $\{s_n\}$ is a Cauchy sequence and hence a convergent sequence.
        \item Is the result in (a) true if we only assume $|s_{n+1} - s_n| < 1/n$ for all $n \in \mathbb{N}$.
    \end{enumerate}
\end{exercise}
