\begin{exercise}
    Prove that every decimal expansion represents a real number.
\end{exercise}

\begin{exercise}
Let $\{s_n\}$ be a sequence.
    \begin{enumerate}
        \item Suppose 
        \begin{align*}
            |s_{n+1} - s_n| < 2^{-n} \quad \text{ for all } \;  n \in \mathbb{N}.
        \end{align*}
        Prove that $\{s_n\}$ is a Cauchy sequence and hence a convergent sequence.
        \item Is the result in (a) true if we only assume $|s_{n+1} - s_n| < 1/n$ for all $n \in \mathbb{N}$.
    \end{enumerate}
\end{exercise}

\begin{exercise}
Prove the Alternating Series Test.
\end{exercise}

\begin{exercise}
    The convergence of a series does not depend on a finite number of terms.  Take two series $\sum\limits_{k=1}^\infty a_k$ and $\sum\limits_{k=1}^\infty b_k$ and suppose the set $\{n\in \mathbb{N}: a_n \neq b_n\}$ is finite.  Prove that either both series diverge or both converge.
\end{exercise}

\begin{exercise}
    Prove that 
    \begin{align}
    \sum\limits_{n=1}^\infty \frac{n^4}{2^n}
    \end{align}
    converges.  Hint: Use induction and the comparison test.
\end{exercise}

\begin{exercise}
    Prove that $\sum\limits_{n=1}^\infty \frac{1}{n(n+1)} = 1$.
\end{exercise}

\begin{exercise}
    Calculate $\sum\limits_{n=1}^\infty \left(\frac23\right)^n$ and $\sum\limits_{n=1}^\infty \left(-\frac23\right)^n$.
\end{exercise}

\textbf{BONUS PROBLEM}
\begin{exercise}

    \end{align*}
\end{exercise}
