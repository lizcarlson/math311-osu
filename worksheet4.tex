\begin{exercise}
    Show that
    \begin{align}
        \sqrt{2}, \sqrt{2\sqrt{2}}, \sqrt{2\sqrt{2\sqrt{2}}}, ...
    \end{align}
    converges and find the limit.
\end{exercise}
\newpage

\begin{exercise}
    Assume the Nested Interval Property, and use a technique similar to the one employed in the proof of the Bolzano-Weierstrass Theorem to give a proof for the Axiom of Completeness.
\end{exercise}
