\begin{exercise}
    Prove that the only sets that are both open and closed are $\mathbb{R}$ and $\emptyset$.
\end{exercise}
\vfill

\begin{exercise}
    Prove or disprove: a finite set is always compact.
\end{exercise}
\vfill \newpage 
\begin{exercise}
Let $K,L \subseteq \mathbb{R}$ be non-empty. Define
$$
d=\inf \{|x-y| : x \in K, y \in L\}.
$$
This turns out to be a "reasonable" definition for the distance between two sets.
\begin{enumerate}
\item If $K$ and $L$ are disjoint and compact, show $d>0$ and that $d=|x_0-y_0|$ for some $x_0 \in K$ 
and $y_0 \in L$. 


\item Prove that it is possible for $d=0$ if we only assume that the disjoint sets $L$ and $K$ are closed.
\end{enumerate}
\end{exercise}
