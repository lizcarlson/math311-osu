    \title{Homework 1}
    \begin{abstract} Solve the following problems.  Due April 8th in class. \end{abstract}
    \maketitle

    \begin{exercise}
            Prove that the infimum of the set $A = \left\{\frac{1}{n} : n \in \mathbb{Z}\right\}$ is $-1$.
    \end{exercise}    

    \begin{exercise}
        Explain the differences between an upper bound, a supremum, and a maximum.  Which definition is the weakest, i.e., has the fewest requirements?  Which definition is the strongest?
    \end{exercise}

    \begin{exercise}
        Let $A$ be a nonempty subset of $\mathbb{R}$ that is bounded above.  Prove that if $\sup(A)$ belongs to $A$, then $\sup(A) = \max(A)$.
    \end{exercise}

    \begin{exercise}
        Let $S$ and $T$ be nonempty subsets of $\mathbb{R}$ with the following property: $s \leq t$ for all $s \in S$ and $t \in T$.
        \begin{enumerate}
            \item Give an example of sets $S$ and $T$ where $S \cap T$ is nonempty.
            \item Give an example of sets $S$ and $T$ where $\sup(S) = \inf(T)$ and $S \cap T$ is the empty set.
        \end{enumerate}
    \end{exercise}

    \begin{exercise}
        Let $x, y \in \mathbb{R}$.  Prove the reverse triangle inequality $||x|- |y|| \leq |x-y|$ using the triangle inequality. 
    \end{exercise}

    \begin{exercise}
        Use the Archimedean Property of $\mathbb{R}$ to prove that $\sup\left\{-\frac{1}{n^2} : n \in \mathbb{N}\right\} = 0$.
    \end{exercise}
